# training error
def RSS(y_hat,y):
    
    return np.sum((y_hat - y)**2)

print('quadratic model training error:', RSS(ys, quad[0]*xs**2 + quad[1]*xs + quad[2]))
print('quintic model training error:', RSS(ys, quint[0]*xs**5 + quint[1]*xs**4 + quint[2]*xs**3 + quint[3]*xs**2 + 
                                           quint[4]*xs + quint[5]))

print('More complex models can lead to comparable/better training error!')
